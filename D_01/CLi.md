# 명령어 환경 실습 코드

- 기본 CLI 명령
- Git 관리 명령

```sh
git --version

git add <file>
git add <dir>
git add .

git commit -m "commit message" // Local Repository

git status

git remote -v

git push [origin] [master] // Remote Repository


git remote add <alias> <repo>

git remote -v

  - origin (fetch)
  - origin
  - <alias> (fetch)
  - alias

git branch -va

git fetch <alias>

git pull <alias> <branch> e.g) git pull yamoo9 master

git push
```
